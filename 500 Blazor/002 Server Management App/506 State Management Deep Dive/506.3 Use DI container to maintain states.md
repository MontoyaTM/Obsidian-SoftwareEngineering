> [!summary] 
> # Core Concept 

This note shows how to **maintain state across multiple Blazor pages using the Dependency Injection (DI) container**, instead of passing data through the URL. It replaces:

- Query strings / route parameters with: a **shared service (`ContainerStorage`) registered in DI**

> [!important] 
> `builder.Services.AddScoped<ContainerStorage>()` Scoped lifetime ensures 
> - Each user/session gets their own state
> - Data is shared across components/pages within that session
> - Prevents cross-user data leakage

| URL State Approach   | DI Container Approach       |
| -------------------- | --------------------------- |
| Uses query params    | Uses shared service         |
| Stateless            | Stateful                    |
| Bookmarkable         | Not bookmarkable            |
| Safer across reloads | Lost on refresh/session end |
| More explicit        | Cleaner code                |

# Program.cs

``` run-csharp
public class Program
{
    public static void Main(string[] args)
    {
        var builder = WebApplication.CreateBuilder(args);
		
        // Add services to the container.
        builder.Services.AddRazorComponents()
            .AddInteractiveServerComponents();
		
        builder.Services.AddTransient<SessionStorage>();
        
        ------------------------------------------------------------------
        // Add ContainerStorage to Scoped Service to prevent shared access
        builder.Services.AddScoped<ContainerStorage>();
        ------------------------------------------------------------------
        
        var app = builder.Build();
		
        // Configure the HTTP request pipeline.
        if (!app.Environment.IsDevelopment())
        {
            app.UseExceptionHandler("/Error");
            // The default HSTS value is 30 days. You may want to change this for production scenarios, see https://aka.ms/aspnetcore-hsts.
            app.UseHsts();
        }
		
        app.UseStatusCodePagesWithReExecute("/not-found", createScopeForStatusCodePages: true);
        app.UseHttpsRedirection();
		
        app.UseAntiforgery();
		
        app.MapStaticAssets();
        app.MapRazorComponents<App>()
            .AddInteractiveServerRenderMode();
		
        app.Run();
    }
}
```

# SetServerName.razor

``` run-csharp
@page "/servername"

@rendermode InteractiveServer

@inject NavigationManager NavigationManager

------------------------------------------------------
// Use the ContainerStorage using dependency injection
@inject ContainerStorage ContainerStorage
------------------------------------------------------

<h3>Server Name</h3>
<br />

@if (!string.IsNullOrWhiteSpace(errorMessage))
{
    <div class="alert alert-danger">
        @errorMessage
    </div>
}

@if(server is not null)
{
    <FieldComponent Label="Name">
        <Control>
            <input type="text" @bind-value="server.Name" class="form-control"></input>
        </Control>
    </FieldComponent>
    <br />
    <button type="button" class="btn btn-primary" @onclick="GoNext">Next</button>
}

@code {
    private Server? server;
	
    private string? errorMessage;
	
	---------------------------------------------
	// Modify method to use void instead of async
    protected override void OnInitialized()
	---------------------------------------------
    {
        base.OnInitialized();
		
        if(RendererInfo.IsInteractive)
        {
            server = new Server();
        }
    }

    private void GoNext()
    {
        if(server is not null)
        {
            if(string.IsNullOrWhiteSpace(server.Name))
            {
                errorMessage = "Server name is required.";
            }
            else
            {
	            ------------------------------------------
	            // Use ContainerStorage's SetServer method
                ContainerStorage.SetServer(server);
	            ------------------------------------------
                NavigationManager.NavigateTo("/cityname");
            }
        }
    }
}
```

# SetStatus.razor

``` run-csharp
@page "/serverstatus"

@rendermode InteractiveServer

@inject NavigationManager NavigationManager

------------------------------------------------------
// Use the ContainerStorage using dependency injection
@inject ContainerStorage ContainerStorage
------------------------------------------------------

<h3>Server Status</h3>
<br />
@if (!string.IsNullOrWhiteSpace(errorMessage))
{
	<div class="alert alert-danger">
		@errorMessage
	</div>
}

@if(server is not null)
{
    <FieldComponent Label="Server Status">
        <Control>
            @if(server.IsOnline)
            {
                <input type="checkbox"
                       @bind-value="server.IsOnline"
                       checked class="form-check-input"></input>
            }
            else
            {
                <input type="checkbox"
                       @bind-value="server.IsOnline"
                       class="form-check-input"></input>
            }

        </Control>
    </FieldComponent>
    <br />

    <button type="button" class="btn btn-primary" @onclick="Save">Save</button>
}

@code {
    private Server? server;

    private string? errorMessage;
	
	----------------------------------------------
	// Modify method to use void instead of async
    protected override void OnInitialized()
    {
        if(RendererInfo.IsInteractive)
        {
            server = ContainerStorage.GetServer();
        }
    }
	----------------------------------------------
	
    private async Task Save()
    {
		if(server is not null)
		{
            ContainerStorage.SetServer(null);
            ServersRepository.AddServer(server);
            NavigationManager.NavigateTo($"/servers/back_from/{server?.City}");
		}
	}
}
```
