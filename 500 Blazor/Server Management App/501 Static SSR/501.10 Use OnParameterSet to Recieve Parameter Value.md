# EditServer.razor

``` run-csharp
@page "/servers/{Id:int}"

<h3>Edit Server</h3>
<br />
<br />

@if(server is not null)
{
    <p>
        @server.Name
    </p>

    <p>
        @server.City
    </p>

    <p>
        @server.IsOnline
    </p>
}

@code {
    [Parameter]
    public int Id { get; set; }

    private Server? server;

    protected override void OnParametersSet()
    {
        server = ServersRepository.GetServerById(Id);
    }
}
```
