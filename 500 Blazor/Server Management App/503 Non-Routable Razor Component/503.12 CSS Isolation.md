``` run-csharp
    <input type="text" class="form-control" placeholder="Search Servers"
           @bind-value="serverFilter"
           @bind-value:event="oninput" />
    <button class="btn btn-outline-secondary" type="button" id="button-search" @onclick="HandleSearch">Search</button>
</div>

@code {
    [Parameter]
    public EventCallback<string> SearchServerCallback { get; set; }

    private string serverFilter = string.Empty;

    private void HandleSearch()
    {
        SearchServerCallback.InvokeAsync(serverFilter);
    }

    public void ClearFilter()
    {
        this.serverFilter = string.Empty;
        StateHasChanged();
    }
}
```

``` run-csharp
@code {
    [Parameter]
    public int Id { get; set; }

    [SupplyParameterFromForm(FormName = "formServer")]
    private Server? server { get; set;  }

    protected override void OnParametersSet()
    {
        server ??= ServersRepository.GetServerById(Id);
    }

    private void Submit()
    {
        if(server is not null)
        {
            ServersRepository.UpdateServer(server.ServerId, server);
        }

        NavigationManager.NavigateTo("/servers");
    }
}
```
