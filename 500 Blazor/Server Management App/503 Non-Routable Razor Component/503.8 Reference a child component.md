# CityListComponent.razor

``` run-csharp
@if (cities is not null && cities.Count > 0)
{
    <div class="container-fluid text-center">
        <div class="row w-50">
            @foreach (var city in cities)
            {
                <CityComponent 
                    city="@city"
                    selectedCity="@selectedCity"
                    SelectCityCallback="HandleCitySelection">
                </CityComponent>
            }
        </div>
    </div>
}

@code {
    [Parameter]
    public EventCallback<string> SelectCityCallback { get; set; }
    
    private List<string> cities = CitiesRepository.GetCities();
    private string selectedCity = "Toronto";

    private void HandleCitySelection(string cityName)
    {
        selectedCity = cityName;
        SelectCityCallback.InvokeAsync(cityName);
    }
```

##### // Method to clear selected city
``` run-csharp
    public void ClearSelection()
    {
        this.selectedCity = string.Empty;
        StateHasChanged();
    }
}
```

# SearchBarComponent.razor

``` run-csharp
<div class="input-group mb-3">
    <input type="text" class="form-control" placeholder="Search Servers"
           @bind-value="serverFilter"
           @bind-value:event="oninput" />
    <button class="btn btn-outline-secondary" type="button" id="button-search" @onclick="HandleSearch">Search</button>
</div>

@code {
    [Parameter]
    public EventCallback<string> SearchServerCallback { get; set; }

    private string serverFilter = string.Empty;

    private void HandleSearch()
    {
        SearchServerCallback.InvokeAsync(serverFilter);
    }
```

##### // Method to clear search bar filter
``` run-csharp
    public void ClearFilter()
    {

    }
}
```
